
#### Criação
```powershell
    npm create vite@latest salaais-new -- --template react-ts
    >> React
    >> Typescript
```

#### Install
```powershell
npm install styled-components axios react-router-dom
npm install react-toastify
npm install --save-dev @types/node
npm install --save-dev @types/styled-components @types/react-router-dom
npm install --save-dev prettier
npm install -D vite-plugin-svgr
npm install --save-dev @vercel/node
npm install -g vercel
npm install vite-plugin-checker --save-dev
npm install -D eslint eslint-plugin-react eslint-plugin-react-hooks @typescript-eslint/eslint-plugin @typescript-eslint/parser eslint-plugin-jsx-a11y eslint-plugin-import
npm install --save-dev @types/google.accounts
npm i -D @types/google-one-tap
```

#### Start
```powershell
npm run dev
vercel dev
vercel logout
vercel login
```
